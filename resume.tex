% --------------------
% Automatic job registry
% --------------------
\makeatletter

\newcommand{\JobRegistry}{}  % internal list

\newcommand{\DeclareJob}[2]{%
  % define the job macro
  \expandafter\newcommand\csname #1\endcsname{#2}%
  % append macro name to registry
  \ifx\JobRegistry\@empty
    \gdef\JobRegistry{#1}%
  \else
    \g@addto@macro\JobRegistry{,#1}%
  \fi
}

\makeatother



% --------------------
% Job rendering engine
% --------------------
\makeatletter

% job renderer
\newcommand{\job}[4]{%
  \begin{tabularx}{\textwidth}{@{}X r@{}}
    \textbf{#1%
      \ifx#2\empty\else, #2\fi
    } & \textit{#3}
  \end{tabularx}\vspace{-6pt}

  \begin{tightitemize}
    #4
  \end{tightitemize}
}

% loop executor
\newcommand{\RenderJobs}{%
  \@for\jobmacro:=\JobRegistry\do{%
    \csname\jobmacro\endcsname
  }%
}



\makeatother
