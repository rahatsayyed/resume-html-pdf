% --- Job Renderer ---
\newcommand{\job}[5]{%
  \textbf{#1}\par\vspace{0pt}
  \begin{tabularx}{\textwidth}{@{}X r@{}}
    {\itshape \ifx#2\empty \else \ifx#3\empty #2 \else \href{#3}{#2} \fi \fi} & {\itshape #4}
  \end{tabularx}\vspace{-10pt}
  \begin{tightitemize} #5 \end{tightitemize}
}

% --- Education Renderer ---
% Usage: \education{Degree}{University}{Dates}{Bullets}
% --- Education Renderer ---
% Usage: \education{Degree}{University}{Dates}{Bullets}
\newcommand{\education}[4]{%
  \textbf{#1}\par\vspace{0pt}
  \begin{tabularx}{\textwidth}{@{}X r@{}}
    {\itshape #2} & {\itshape #3}
  \end{tabularx}
  \ifx#4\empty
  % no bullets
  \else
    \vspace{-20pt}
    \begin{tightitemize} #4 \end{tightitemize}
  \fi
}

% --- Certification Renderer ---
% Usage: \certification{Name}{Issuer/Date}{Bullets}
\newcommand{\certification}[3]{%
\begin{tabularx}{\textwidth}{@{}X r@{}}
  \textbf{#1}\par\vspace{0pt}
    {\itshape \empty} & {\itshape #2}
  \end{tabularx}\vspace{0pt}
  \ifx#3\empty
  % no bullets
  \else
    \vspace{-20pt}
    \begin{tightitemize} #3 \end{tightitemize}
    \fi
}

\newcommand{\RenderCertifications}{%

}
