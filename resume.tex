\newcommand{\DeclareHeaderItem}[3]{%
  % store renderer macro
  \expandafter\newcommand\csname header@#1\endcsname{%
    \ifx#3\empty
      #2%
    \else
      \href{#3}{#2}%
    \fi
  }%
  % append to registry
  \ifx\HeaderRegistry\@empty
    \gdef\HeaderRegistry{header@#1}%
  \else
    \g@addto@macro\HeaderRegistry{,header@#1}%
  \fi
}

\newcommand{\DeclareJob}[2]{%
  \expandafter\newcommand\csname #1\endcsname{#2}%
  \ifx\JobRegistry\@empty \gdef\JobRegistry{#1}\else \g@addto@macro\JobRegistry{,#1}\fi
}

\newcommand{\DeclareEducation}[2]{%
  \expandafter\newcommand\csname #1\endcsname{#2}%
  \ifx\EducationRegistry\@empty \gdef\EducationRegistry{#1}\else \g@addto@macro\EducationRegistry{,#1}\fi
}

\newcommand{\DeclareCertification}[2]{%
  \expandafter\newcommand\csname #1\endcsname{#2}%
  \ifx\CertificationRegistry\@empty \gdef\CertificationRegistry{#1}\else \g@addto@macro\CertificationRegistry{,#1}\fi
}

% --- Job Renderer ---
\newcommand{\job}[5]{%
  % ---- Role ----
  {\BodySize\bfseries #1}\par

  % ---- Company + Dates ----
  {\BodySize\itshape
  \ifx#2\empty
  %
  \else
  \ifx#3\empty
  #2
  \else
  \href{#3}{#2}
  \fi
  \fi
  \hfill
  #4
  }\par
  \vspace{-10pt}
  % ---- Bullets ----
  \begin{tightitemize}
    #5
  \end{tightitemize}

  \vspace{\JobGap}
}


% --- Education Renderer ---
% Usage: \education{Degree}{University}{Dates}{Bullets}
\newcommand{\education}[4]{%
  {\BodySize\bfseries #1}\par
  {\BodySize\itshape #2 \hfill #3}\par

  \ifx#4\empty
  \else
    \begin{tightitemize}
      #4
    \end{tightitemize}
  \fi

  \vspace{\JobGap}
}


% --- Certification Renderer ---
% Usage: \certification{Name}{Issuer/Date}{Bullets}
\newcommand{\certification}[3]{%
  {\BodySize \textbf{#1} \hfill \itshape #2}\par

  \ifx#3\empty
  \else
    \begin{tightitemize}
      #3
    \end{tightitemize}
  \fi

  \vspace{2pt}
}



% --- Loop Executors ---
\newcount\HeaderItemIndex

\newcommand{\RenderHeaderItems}{%
  \HeaderItemIndex=0
  \@for\headermacro:=\HeaderRegistry\do{%
    \ifnum\HeaderItemIndex=0
      % First item, do nothing
    \else
      \ifnum\HeaderItemIndex=4
        \\[3pt] % Line break before the 4th item (index 3)
      \else
        \sep % Separator otherwise
      \fi
    \fi
    \csname\headermacro\endcsname
    \advance\HeaderItemIndex by 1
  }%
}



\newcommand{\RenderJobs}{%
  \@for\jobmacro:=\JobRegistry\do{\csname\jobmacro\endcsname}%
}

\newcommand{\RenderEducation}{%
  \@for\edmacros:=\EducationRegistry\do{\csname\edmacros\endcsname}%
}

\newcommand{\RenderCertifications}{%
  \@for\certmacro:=\CertificationRegistry\do{\csname\certmacro\endcsname}%
}
